function broadcastLeaderboard(data) {
  wss.clients.forEach(client => {
    if (client.readyState === 1) {
      client.send(JSON.stringify(data));
    }
  });
}

app.post("/api/ai", async (req, res) => {
  const { cmd, context } = req.body;

  const prompt = `
You are a senior SRE mentor.

User typed:
${cmd}

System state:
${JSON.stringify(context)}

Give a short hint, NOT the solution.
`;

  const response = await axios.post(
    "https://api.openai.com/v1/chat/completions",
    {
      model: "gpt-4o-mini",
      messages: [{ role: "user", content: prompt }]
    },
    {
      headers: {
        Authorization: `Bearer ${process.env.OPENAI_API_KEY}`
      }
    }
  );

  res.json({
    hint: response.data.choices[0].message.content
  });
});

async function replay(id) {
  const res = await fetch(`/api/replay/${id}`);
  const events = await res.json();

  for (const e of events) {
    await new Promise(r => setTimeout(r, 800));
    print(`$ ${e.cmd}`);
    print(e.output);
  }
}

import { useState, useEffect } from "react";
import { getScenarios } from "./api";
import ScenarioEditor from "./ScenarioEditor";

export default function AdminApp() {
  const [scenarios, setScenarios] = useState([]);
  const [active, setActive] = useState(null);

  useEffect(() => {
    getScenarios().then(setScenarios);
  }, []);

  if (active) {
    return <ScenarioEditor id={active} onBack={() => setActive(null)} />;
  }

  return (
    <div style={{ padding: 20 }}>
      <h2>🧠 WinLab Admin</h2>

      {scenarios.map(s => (
        <div
          key={s.id}
          onClick={() => setActive(s.id)}
          style={{
            padding: 12,
            border: "1px solid #ddd",
            marginBottom: 8,
            cursor: "pointer"
          }}
        >
          {s.title} (v{s.version})
        </div>
      ))}
    </div>
  );
}
✍️ 3. SCENARIO EDITOR (core Notion-like)
// admin/ScenarioEditor.jsx
import { useState, useEffect } from "react";
import { getScenario, updateScenario } from "./api";
import TranslationPanel from "./TranslationPanel";

export default function ScenarioEditor({ id, onBack }) {
  const [data, setData] = useState(null);
  const [title, setTitle] = useState("");

  useEffect(() => {
    getScenario(id).then(d => {
      setData(d);
      setTitle(d.scenario.title);
    });
  }, [id]);

  if (!data) return "loading...";

  async function save() {
    await updateScenario({ id, title });
    alert("Saved + version++");
  }

  return (
    <div style={{ padding: 20 }}>
      <button onClick={onBack}>← Back</button>

      <input
        value={title}
        onChange={e => setTitle(e.target.value)}
        style={{
          fontSize: 24,
          fontWeight: "bold",
          border: "none",
          outline: "none",
          width: "100%"
        }}
      />

      <h3>⚙️ Commands</h3>
      {data.commands.map(cmd => (
        <div key={cmd.id} style={{ marginBottom: 8 }}>
          <code>{cmd.command}</code> → {cmd.response_key}
        </div>
      ))}

      <TranslationPanel
        scenarioId={id}
        translations={data.translations}
      />

      <button onClick={save}>💾 Save</button>
    </div>
  );
}
🌍 4. TRANSLATION PANEL (inline editing)
// admin/TranslationPanel.jsx
import { useState } from "react";
import { updateTranslation } from "./api";

export default function TranslationPanel({ scenarioId, translations }) {
  const [lang, setLang] = useState("en");

  const filtered = translations.filter(t => t.lang === lang);

  async function save(key, value) {
    await updateTranslation({
      scenarioId,
      lang,
      key,
      value
    });
  }

  return (
    <div>
      <h3>🌍 Translations</h3>

      <select onChange={e => setLang(e.target.value)}>
        <option value="en">EN</option>
        <option value="it">IT</option>
      </select>

      {filtered.map(t => (
        <div key={t.id} style={{ marginBottom: 8 }}>
          <div style={{ fontSize: 12, color: "#666" }}>{t.key}</div>

          <input
            defaultValue={t.value}
            onBlur={e => save(t.key, e.target.value)}
            style={{ width: "100%" }}
          />
        </div>
      ))}
    </div>
  );
}
