Ho implementato uno scheletro TypeScript production-ready per i due flussi che hai descritto: Identity/User Lifecycle Email Flow + Alert Flow con severità, deduplica e rate limiting. La configurazione progetto (package.json + tsconfig) è stata aggiunta per supportare build/check tipizzati. 

Ho aggiunto un servizio email centralizzato con Resend, fallback sicuro se la chiave API non è configurata, e campi env per APP_BASE_URL, RESEND_API_KEY, EMAIL_FROM. 

Ho implementato il lifecycle email completo P0:

verify email (sendVerifyEmail)

password reset con expiry 15 minuti (sendPasswordResetEmail)

new device login (sendNewDeviceLoginEmail)

account deleted (sendAccountDeletedEmail). 

Ho implementato la parte sicurezza token come richiesto: token raw generato, hash SHA-256 salvato, mai token in chiaro in storage. 

Ho implementato l’Alert Dispatcher con:

deduplicazione finestra 5 minuti

rate limit 5 email/ora/utente

invio email solo se CRITICAL e solo se permesso dalle preferenze utente. 

Ho aggiunto un event bus tipizzato e bootstrap del flow SERVICE_DOWN -> dispatchAlert, pronto per integrazione con il tuo monitoring engine/event bus. 

Ho documentato rapidamente architettura e note operative nel README.

# MindfulMotivation – Identity Email + Alert Flow Skeleton

Production-oriented TypeScript skeleton that covers:

- User lifecycle email events (verify, reset, new device login, account deleted)
- Token security (hashed token storage, short reset expiry)
- Alert engine with deduplication and per-user rate limiting
- CRITICAL-only email escalation with user notification preferences

## Main Files

- `src/services/userLifecycleEmailFlow.ts`
- `src/core/alertDispatcher.ts`
- `src/core/eventBus.ts`

## Notes

- `inMemoryDb` is intentionally a stub; replace with your real persistence layer.
- Configure `RESEND_API_KEY`, `APP_BASE_URL`, and `EMAIL_FROM` in your environment.

import { sendEmail } from '../services/emailService.js';
import type { AlertEvent, User } from '../types/domain.js';

const DEDUP_WINDOW_MS = 5 * 60 * 1000;
const RATE_LIMIT_WINDOW_MS = 60 * 60 * 1000;
const RATE_LIMIT_MAX = 5;

const latestAlertByKey = new Map<string, number>();
const emailSendsByUser = new Map<string, number[]>();

function sameAlertInLastFiveMinutes(alert: AlertEvent): boolean {
  const last = latestAlertByKey.get(alert.key);
  if (!last) {
    latestAlertByKey.set(alert.key, alert.createdAt);
    return false;
  }

  if (alert.createdAt - last < DEDUP_WINDOW_MS) {
    return true;
  }

  latestAlertByKey.set(alert.key, alert.createdAt);
  return false;
}

function allowEmailByRateLimit(userId: string, now: number): boolean {
  const sends = emailSendsByUser.get(userId) ?? [];
  const recent = sends.filter((timestamp) => now - timestamp <= RATE_LIMIT_WINDOW_MS);

  if (recent.length >= RATE_LIMIT_MAX) {
    emailSendsByUser.set(userId, recent);
    return false;
  }

  recent.push(now);
  emailSendsByUser.set(userId, recent);
  return true;
}

export async function dispatchAlert(alert: AlertEvent, user: User): Promise<void> {
  if (sameAlertInLastFiveMinutes(alert)) {
    return;
  }

  // Placeholder: broadcast to UI timeline/toast channel.
  // In browser runtime you can replace with window.dispatchEvent(new CustomEvent('ALERT', { detail: alert }))
  console.info('[ALERT_TIMELINE]', alert);

  const shouldEmail =
    user.notifyByEmail &&
    user.alertSeverities.includes(alert.severity) &&
    alert.severity === 'CRITICAL' &&
    allowEmailByRateLimit(user.id, alert.createdAt);

  if (!shouldEmail) {
    return;
  }

  await sendEmail(
    user.email,
    `🚨 Critical Alert: ${alert.message}`,
    `<h2>Critical Incident</h2><p>${alert.message}</p><p>Time: ${new Date(alert.createdAt).toISOString()}</p>`
  );
}

type EventHandler<Payload> = (payload: Payload) => void | Promise<void>;

export class EventBus<Events extends Record<string, unknown>> {
  private handlers: { [K in keyof Events]?: EventHandler<Events[K]>[] } = {};

  on<K extends keyof Events>(event: K, handler: EventHandler<Events[K]>): void {
    const list = this.handlers[event] ?? [];
    list.push(handler);
    this.handlers[event] = list;
  }

  async emit<K extends keyof Events>(event: K, payload: Events[K]): Promise<void> {
    const list = this.handlers[event] ?? [];
    for (const handler of list) {
      await handler(payload);
    }
  }
}

import { env } from '../config/env.js';
import { db } from './inMemoryDb.js';
import { sendEmail } from './emailService.js';
import { generateRawToken, hashToken } from './tokenService.js';
import type { User } from '../types/domain.js';

const VERIFY_TOKEN_TTL_MS = 24 * 60 * 60 * 1000;
const RESET_TOKEN_TTL_MS = 15 * 60 * 1000;

export async function sendVerifyEmail(user: User): Promise<void> {
  const token = generateRawToken();
  const tokenHash = hashToken(token);

  await db.saveVerifyToken({
    userId: user.id,
    tokenHash,
    expiresAt: Date.now() + VERIFY_TOKEN_TTL_MS
  });

  const link = `${env.appBaseUrl}/verify?token=${token}`;
  await sendEmail(
    user.email,
    'Verify your email',
    `<p>Welcome!</p><p>Please verify your email: <a href="${link}">Verify account</a>.</p>`
  );
}

export async function sendPasswordResetEmail(user: User): Promise<void> {
  const token = generateRawToken();
  const tokenHash = hashToken(token);

  await db.saveResetToken({
    userId: user.id,
    tokenHash,
    expiresAt: Date.now() + RESET_TOKEN_TTL_MS
  });

  const link = `${env.appBaseUrl}/reset-password?token=${token}`;
  await sendEmail(
    user.email,
    'Reset your password',
    `<p>Reset requested.</p><p><a href="${link}">Reset password</a> (expires in 15 minutes).</p>`
  );
}
