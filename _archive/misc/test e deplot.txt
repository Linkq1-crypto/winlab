export async function apiFetch(path: string, options?: RequestInit) {
  try {
    const res = await fetch(API_BASE + path, options);

    if (!res.ok) throw new Error('API error');

    return await res.json();

  } catch (err) {
    console.warn('⚠️ API failed → using MOCK');

    return mockFetch(path, options);
  }
}

import { eventBus } from './eventBus';

class ChaosEngine {
  private interval: any;

  start() {
    this.interval = setInterval(() => {
      const events = [
        'API_FAILURE',
        'HIGH_LATENCY',
        'VLAN_DOWN',
        'DEVICE_NON_COMPLIANT',
        'SERVICE_DOWN'
      ];

      const random = events[Math.floor(Math.random() * events.length)];

      console.log('🔥 CHAOS:', random);

      eventBus.emit(random);

    }, 3000);
  }

  stop() {
    clearInterval(this.interval);
  }
}

export const chaosEngine = new ChaosEngine();

🧠 4. HEALTH SCORE (🔥 importantissimo)

Aggiungi:

function computeHealth(m) {
  let score = 100;

  score -= m.errors * 5;
  score -= m.apiFailures * 3;

  if (m.latency.length > 0) {
    const avg = m.latency.reduce((a, b) => a + b, 0) / m.latency.length;
    if (avg > 1000) score -= 10;
  }

  return Math.max(score, 0);
}

import { useMonitoring } from './monitoringStore';

export function startMonitoringSampler() {
  setInterval(() => {
    const s = useMonitoring.getState();

    const avgLatency =
      s.latency.length > 0
        ? s.latency.reduce((a, b) => a + b, 0) / s.latency.length
        : 0;

    useMonitoring.getState().pushPoint({
      time: Date.now(),
      latency: avgLatency,
      errors: s.errors,
      api: s.apiCalls
    });

  }, 500);
}

import { create } from 'zustand';

type Incident = {

};

export function analyzeRootCause() {
  const events = useTimeline.getState().events;

  const lastCritical = events.find(e => e.severity === 'CRITICAL');

  if (!lastCritical) return 'No critical incidents';

  const index = events.indexOf(lastCritical);

  const context = events.slice(index, index + 5);

  // semplice euristica (espandibile)
  if (context.some(e => e.type === 'HIGH_LATENCY')) {
    return 'Root cause: latency spike → system degradation';
  }

  if (context.some(e => e.type === 'API_FAILURE')) {
    return 'Root cause: backend/API failure';
  }

  if (context.some(e => e.type === 'VLAN_DOWN')) {
    return 'Root cause: network failure';
  }

  return 'Root cause: unknown';
}

import { graph } from './dependencyGraph';
import { eventBus } from './eventBus';

export function triggerFailure(nodeId: string) {
  const node = graph.getNode(nodeId);

  if (!node) return;

  node.status = 'DOWN';

  eventBus.emit('SERVICE_DOWN', { node: nodeId });

  propagate(nodeId);
}

function propagate(failedId: string) {
  const dependents = graph.getDependents(failedId);

  for (const dep of dependents) {
    if (dep.status === 'UP') {
      dep.status = 'DEGRADED';

      eventBus.emit('SERVICE_DEGRADED', { node: dep.id });

      // cascading
      propagate(dep.id);
    }
  }
}

export type ServiceStatus = 'UP' | 'DEGRADED' | 'DOWN' | 'RECOVERING';

export type ServiceNode = {
  id: string;
  status: ServiceStatus;
  dependsOn: string[];
  tier?: 'frontend' | 'application' | 'data' | 'network';
  criticality?: 1 | 2 | 3;
};

class DependencyGraph {
  private nodes = new Map<string, ServiceNode>();

  addNode(node: ServiceNode) {
    this.nodes.set(node.id, node);
  }

  getNode(id: string) {
    return this.nodes.get(id);
  }

  updateStatus(id: string, status: ServiceStatus) {
    const node = this.nodes.get(id);
    if (!node) return;
    node.status = status;
  }

  getDependents(id: string) {
    return [...this.nodes.values()].filter((n) => n.dependsOn.includes(id));
  }

  list() {
    return [...this.nodes.values()];
  }

  reset(status: ServiceStatus = 'UP') {
    for (const node of this.nodes.values()) node.status = status;
  }
}

export const graph = new DependencyGraph();

export function seedDefaultTopology() {
  graph.addNode({ id: 'DB', status: 'UP', dependsOn: [], tier: 'data', criticality: 1 });
  graph.addNode({ id: 'Auth', status: 'UP', dependsOn: ['DB'], tier: 'application', criticality: 1 });
  graph.addNode({ id: 'API', status: 'UP', dependsOn: ['DB', 'Auth'], tier: 'application', criticality: 1 });
  graph.addNode({ id: 'Frontend', status: 'UP', dependsOn: ['API'], tier: 'frontend', criticality: 2 });
}

export type EventHandler<T = unknown> = (payload: T) => void;

class EventBus {
  private listeners = new Map<string, Set<EventHandler>>();

  on<T = unknown>(event: string, handler: EventHandler<T>) {
    const set = this.listeners.get(event) ?? new Set<EventHandler>();
    set.add(handler as EventHandler);
    this.listeners.set(event, set);

    return () => this.off(event, handler as EventHandler);
  }

  off(event: string, handler: EventHandler) {
    const set = this.listeners.get(event);
    if (!set) return;
    set.delete(handler);
    if (set.size === 0) this.listeners.delete(event);
  }

  emit<T = unknown>(event: string, payload?: T) {
    const set = this.listeners.get(event);
    if (!set) return;
    for (const handler of set) handler(payload);
  }
}

export const eventBus = new EventBus();

    eventBus.emit.bind(eventBus);

import { logIncident } from './timelineStore';

export function triggerFailure(serviceId: string) {
  const node = graph.getNode(serviceId);
  if (!node) return;

  graph.updateStatus(serviceId, 'DOWN');
  logIncident('INFRA', `${serviceId} is DOWN`, 'CRITICAL', { serviceId });
  eventBus.emit('SERVICE_DOWN', { serviceId });

  cascade(serviceId);
}

function cascade(failedId: string) {
  const impacted = graph.getDependents(failedId);

  for (const dependent of impacted) {
    if (dependent.status === 'DOWN') continue;

    graph.updateStatus(dependent.id, 'DEGRADED');
    logIncident('INFRA', `${dependent.id} degraded due to ${failedId}`, 'WARN', {
      serviceId: dependent.id,
      cause: failedId,
    });
    eventBus.emit('SERVICE_DEGRADED', { serviceId: dependent.id, cause: failedId });

    cascade(dependent.id);
  }
}

export function recommendRemediation(): RemediationAction[] {

  const nodes = graph.list();

  return nodes
    .filter((n) => n.status === 'DOWN' || n.status === 'DEGRADED')
    .map((n) => {
      const related = incidents.filter((i) =>
        i.metadata?.serviceId === n.id || i.metadata?.cause === n.id
      );

      const hasLatency = related.some((r) => r.message.toLowerCase().includes('latency'));
      const hasDbSignal = n.id === 'DB' || related.some((r) => String(r.metadata?.cause) === 'DB');

      if (hasDbSignal) {

          serviceId: n.id,
          action: 'FAILOVER',
          reason: 'Dipendenza da data layer in stato critico',
          confidence: 0.88,

import { eventBus } from './eventBus';
import { graph } from './dependencyGraph';
import { logIncident } from './timelineStore';
import { recommendRemediation } from './remediationAdvisor';

class SelfHealingEngine {
  private timer: ReturnType<typeof setInterval> | null = null;

  start(intervalMs = 1500) {
    if (this.timer) return;

    this.timer = setInterval(() => {
      const actions = recommendRemediation();

      for (const action of actions) {
        const target = graph.getNode(action.serviceId);
        if (!target) continue;

        // Simplified remediation workflow for demo/simulation.
        if (action.action === 'FAILOVER' || action.action === 'RESTART') {
          graph.updateStatus(target.id, 'RECOVERING');
          eventBus.emit('REMEDIATION_STARTED', action);
          logIncident('REMEDIATION', `${action.action} started on ${target.id}`, 'WARN', {
            serviceId: target.id,
            action: action.action,
            confidence: action.confidence,
          });

          setTimeout(() => {
            graph.updateStatus(target.id, 'UP');
            eventBus.emit('REMEDIATION_SUCCESS', action);
            logIncident('REMEDIATION', `${target.id} recovered via ${action.action}`, 'INFO', {
              serviceId: target.id,
            });
          }, 1200);
        }
      }
    }, intervalMs);
  }

  stop() {
    if (!this.timer) return;
    clearInterval(this.timer);
    this.timer = null;
  }
}

export const selfHealingEngine = new SelfHealingEngine();

  w.startSelfHealing = () => selfHealingEngine.start();
  w.stopSelfHealing = () => selfHealingEngine.stop();
}

export type Severity = 'INFO' | 'WARN' | 'CRITICAL';

export type Incident = {
  id: number;
  type: string;
  message: string;
  severity: Severity;
  ts: number;
  metadata?: Record<string, unknown>;
};

class TimelineStore {
  private events: Incident[] = [];

  add(input: Omit<Incident, 'id' | 'ts'> & Partial<Pick<Incident, 'id' | 'ts'>>) {
    const incident: Incident = {
      id: input.id ?? Date.now() + Math.floor(Math.random() * 10_000),
      ts: input.ts ?? Date.now(),
      type: input.type,
      message: input.message,
      severity: input.severity,
      metadata: input.metadata,
    };

    this.events = [incident, ...this.events].slice(0, 200);
    return incident;
  }

  all() {
    return [...this.events];
  }

  latest() {
    return this.events[0];
  }
}

export const timelineStore = new TimelineStore();

export function logIncident(type: string, message: string, severity: Severity, metadata?: Record<string, unknown>) {
  return timelineStore.add({ type, message, severity, metadata });
}
